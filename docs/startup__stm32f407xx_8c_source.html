220void Reset_Handler(void)
221{
222 /* copy the .data section to SRAM */
223 uint32_t size = (uint32_t)&_edata - (uint32_t)&_sdata;
224
225 uint8_t *pDst = (uint8_t *)&_sdata; /* SRAM */
226 uint8_t *pSrc = (uint8_t *)&_la_data; /* Flash */
227
228 for (uint32_t i = 0; i < size; i++){
229 *pDst++ = *pSrc++;
230 }
231
232 /* Initialise the .bss section to 0 in SRAM */
233 size = (uint32_t)&_ebss - (uint32_t)&_sbss;
234 pDst = (uint8_t *)&_sbss;
235
236 for (uint32_t i = 0; i < size; i++){
237 *pDst++ = 0;
238 }
239
240 __libc_init_array();
241
242 main();
243}
